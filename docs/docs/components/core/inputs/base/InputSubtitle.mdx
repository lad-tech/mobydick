## Example

```tsx
import {
  InputSubtitle,
  IInputsTypes,
} from '@lad-tech/mobydick-core';

const InputList = () => {

  return <InputSubtitle type={IInputsTypes.disabled} subtitle={'Subtitle'} />
};

export default InputList;

```
